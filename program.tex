\section{Блок диаграммы. Вершинная функция.}

\begin{figure}[h!]
\centering
\includegraphics[scale=0.6]{ladder.eps}
\caption{Блок лестничной диаграммы}
\label{ladder}
\end{figure}

Рассматриваем лестничные ряды, составленные из изображенных на рисунке \ref{ladder} блоков.
В статьях по дуальной технике в узлах диаграмм стоят вершинные части примесной задачи, а линиями служат дуальные функции Грина.
Нам будет удобнее в узлах разместить неприводимые части функций Грина, а линии, соответственно, домножить на $g^{-2}$.

Таким образом, ``четырехугольный'' узел равен
\begin{equation}
\begin{split}
g^{(2irr)}_{1234}
 &= \left<c^\dag_\downarrow(t+\tau) c_\uparrow(t) c^\dag_\uparrow(\tau') c_\downarrow(0)\right> \\
 &- \cancel{\left<c^\dag_\downarrow(t+\tau) c_\uparrow(t)\right>\left< c^\dag_\uparrow(\tau') c_\downarrow(0)\right>} \\
 &+ \left<c^\dag_\downarrow(t+\tau) c_\downarrow(0)\right>\left< c^\dag_\uparrow(\tau') c_\uparrow(t)\right>
\end{split}
\end{equation}

Линия, в нулевом приближении, равна
\begin{equation}
\begin{split}
&\tilde{G}_{\omega k} = g^{-2}_\omega \tilde{G}^{(0)}_{\omega k} = (g_\omega-\epsilon^{-1}_k)^{-1},\;\text{где} \\
g_\omega &= \int_0^\beta \mathrm{d}\tau \left< c^\dag_\uparrow(\tau)c_\uparrow(0) \right> e^{i\omega\tau} = \frac{i\omega}{\omega^2+\mu^2} \\
\epsilon_k &= 2 t (\cos k_x+\cos k_y).
\end{split}
\end{equation}

Лестница в целом характеризуется волновым вектором $K$ и частотой $\Omega$. ``Внутренние'' индексы 
будем использовать во временн{\'o}м представлении.

Таким образом, нам понадобятся величины 
\begin{equation}
X^{(0)}_{\Omega K \tau} = \beta^{-1} \iint_{-\pi}^{\pi} \mathrm{d}^2k \sum_\omega \tilde{G}_{\omega k} \tilde{G}_{\substack{\Omega+\omega\\K+k}} e^{-i \omega \tau},
\end{equation}
где суммирование идёт по мацубаровским частотам $\omega_n = \frac{(2n+1)\pi}{\beta}, n\in\mathbb{Z}$, а интегрирование по 1-ой зоне Бриллюэна.

\begin{equation}
g^{(2irr)}_{\Omega \tau \tau'} = \int_0^\beta g^{(2irr)}_{ t+\tau\downarrow, t\uparrow,\tau'\uparrow,0\downarrow}  e^{i\Omega t} dt.
\end{equation}

Обсуждаемый блок представляет собой свертку
\begin{equation}
\int_0^\beta  X^{(0)}_{\Omega K \tau-\tau_1} g^{(2irr)}_{\Omega \tau_1 \tau'} d\tau_1
\end{equation}

{\bf Задача 1}. Хочется посмотреть на 3D график $g^{(2irr)}_{\Omega \tau \tau'}$ при $\Omega=0$, и на график $X^{(0)}_{\Omega K \tau}$ при $\Omega=0$ для нескольких характерных $K$. 

\begin{figure}[h!]
\centering
\includegraphics[scale=0.6]{g2irr.eps}
\caption[$g^{(2irr)}_{0 \tau \tau'}$]{3D график $g^{(2irr)}_{0 \tau \tau'}$}
\end{figure}

Легко заметить что $\tilde{G}_{\omega k}\xrightarrow[\omega\to\pm\infty]{}-\epsilon_k$. Поэтому определим $\tilde{X}^{(0)}$

\begin{equation}
\begin{split}
\tilde{X}^{(0)}_{\Omega K \tau} &= X^{(0)}_{\Omega K \tau} - \beta^{-1} \iint_{-\pi}^{\pi} \mathrm{d}^2k \sum_\omega \epsilon_k \epsilon_{k+K} e^{-i \omega \tau} \\
                                &= X^{(0)}_{\Omega K \tau} - 4\pi^2 t \frac{\epsilon_K}{\beta} \delta(\tau) 
\end{split}
\end{equation}

Таким образом мы получим сходящуюся сумму по $\omega$.

\begin{equation}
X_{\substack{\Omega K\\ \omega\omega'}} = \chi_{\substack{\Omega\\ \omega\omega'}} 
  + \frac{1}{\beta}\sum_{\omega''}\chi_{\substack{\Omega\\ \omega\omega''}}
    \overbrace{\left(\int_k \tilde{G}_{\substack{\Omega+\omega''\\K+k}}\tilde{G}_{\substack{\omega''\\k}}\right)}^{-\chi^{(0)}_{\substack{\Omega K\\\omega''}}}
    X_{\substack{\Omega K\\\omega''\omega'}}
\end{equation}

Делаем Фурье преобразования

\begin{equation}
\begin{split}
X_{\substack{\Omega K\\ \tau\tau'}} &= 
  \chi_{\substack{\Omega\\ \tau\tau'}} - \sum_{\omega\omega'\omega''}\chi_{\substack{\Omega\\\omega\omega''}}\chi^{(0)}_{\substack{\Omega K\\\omega''}}X_{\substack{\Omega K\\\omega''\omega'}}e^{-i\omega\tau}e^{-i\omega'\tau'}\\
 &= \chi_{\substack{\Omega\\ \tau\tau'}} - \sum_{\omega''}\chi_{\substack{\Omega\\\tau\omega''}}\chi^{(0)}_{\substack{\Omega K\\\omega''}}X_{\substack{\Omega K\\\omega''\tau'}}\\
 &= \chi_{\substack{\Omega\\ \tau\tau'}} - \sum_{\omega''}\chi_{\substack{\Omega\\\tau\omega''}}\chi^{(0)}_{\substack{\Omega K\\\omega''}}\left(\int_{\tau''}X_{\substack{\Omega K\\\tau''\tau'}}e^{i\omega''\tau''}\right)\\
 &= \chi_{\substack{\Omega\\ \tau\tau'}} - \int_{\tau''}\left(\sum_{\omega''}\chi_{\substack{\Omega\\\tau\omega''}}\chi^{(0)}_{\substack{\Omega K\\\omega''}}e^{i\omega''\tau''}\right)X_{\substack{\Omega K\\\tau''\tau'}}\\
 \sum_{\omega''}\chi_{\substack{\Omega\\\tau\omega''}}\chi^{(0)}_{\substack{\Omega K\\\omega''}}e^{i\omega''\tau''}
  &= \sum_{\omega''}\left(\int_{\tilde{\tau}''}\chi_{\substack{\Omega\\\tau\tilde{\tau}''}}e^{i\omega''\tilde{\tau}''}\right)\chi^{(0)}_{\substack{\Omega K\\\omega''}}e^{i\omega''\tau''}
   = \int_{\tilde{\tau}''}\chi_{\substack{\Omega\\\tau\tilde{\tau}''}}\chi^{(0)}_{\substack{\Omega K\\-\tilde{\tau}''-\tau''}}
\end{split}
\end{equation}

Получаем интегральное уравнение

\begin{equation}
\begin{split}
X_{\substack{\Omega K\\ \tau\tau'}} + 
  \int_0^\beta\mathrm{d}\tau''\underbrace{\left(\int_{\tilde{\tau}''}\chi_{\substack{\Omega\\\tau\tilde{\tau}''}}\chi^{(0)}_{\substack{\Omega K\\-\tilde{\tau}''-\tau''}}\right)}_{A_{\tau\tau''}}X_{\substack{\Omega K\\\tau''\tau'}}
  = \chi_{\substack{\Omega\\ \tau\tau'}}
\end{split}
\end{equation}

Образуем сетку $N\times N$ по $\tau$-переменным: $\tau_i=i\Delta\tau, i\in \overline{0,N}$, где $\Delta\tau=\frac{\beta}{N}$, тогда интегральное уравнение
можно будет заменить на систему $(N+1)^2$ линейных уравнений c $(N+1)^2$ неизвестными

\begin{equation}
 X_{ij} + \sum_{k=0}^N a_{ik}X_{kj} = b_{ij}
\end{equation}

где $X_{ij} = X_{\substack{\Omega K\\ \tau_i\tau_j}}, a_{ik} = \int_0^\beta\mathrm{d}{\tilde{\tau}''}\chi_{\substack{\Omega\\\tau_i\tilde{\tau}''}}\chi^{(0)}_{\substack{\Omega K\\-\tilde{\tau}''-\tau_k}},b_{ij}=\chi_{\substack{\Omega\\ \tau_i\tau_j}}$
